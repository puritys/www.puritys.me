Json to Array 格式转换

Tags:
  • Javascript
  • Array
繁中版

刚刚在 Javascript 想要把 Array 转成 Json string ，上网找一些别人写好的来用，结果每个都失败，有些连跑都跑不过，不然就是 Array key 不见了，变成 index , 0 1 2 3的索引，真是气死人，后来花了 5 分钟自已写一个，还是自已写的最顺手。

Array 转换成 Json String

Example 
  1. function jsonToString(arr) {
  2.    var s="";
  3.    if( arr instanceof Array || arr instanceof Object){
  4.       var isObj=0;
  5.       //check value type
  6.       for(key in arr){
  7.          if( isNaN(parseInt(key)) ){ //key is string
  8.             isObj=1;
  9.          }
  10.          else{
  11.             //key is index , check sort
  12.             var na=arr.length;
  13.             var tmp=arr;
  14.                          //hack for ie
  15.             arr=Array();
  16.             for(var j=0;j<na;j++){
  17.                if( typeof(tmp[j])=="undefined" ){
  18.                   arr[j]="";
  19.                }
  20.                else{
  21.                   arr[j]=tmp[j];
  22.                }
  23.             }
  24.          }
  25.          break;
  26.       }
  27.       
  28.       for(key in arr){
  29.          var value=arr[key];
  30.          if( isObj ){
  31.             if(s){s+=',';}
  32.             s+='"'+key+'":'+jsonToString(value);
  33.          }
  34.          else{
  35.             if(s){s+=',';}
  36.             s+=jsonToString(value);
  37.          }
  38.       }
  39.       if(isObj)
  40.          s='{'+s+'}';
  41.       else
  42.          s='['+s+']'
  43.    }
  44.    else{
  45.       if(!isNaN(parseInt(arr))){
  46.          s+=arr;
  47.       }
  48.       else{
  49.          s='"'+arr+'"';
  50.       }
  51.    }
  52.    return s;
  53. }

测试有 key 的 Array , 索引值为数字的 Array

Example 
  1. var a=Array();
  2. a["a"]=Array("2","3","40");
  3. a["b"]=5;
  4. debug(jsonToString(a));
  5. //output: {"a":[2,3,40],"b":5}
  6.  
  7. b=Array();
  8. b[0]=Array("2","3","40");
  9. b[1]=5;
  10. debug(jsonToString(b));
  11. //output: [[2,3,40],5]
  12.  
  13. debug(stringToJson('{"a":[2,3,40],"b":5}'));
  14. /*output 
  15. Array{
  16.     a = 2,3,40
  17.     b = 5
  18. }
  19. */

Example

  3. b[2]=10;//漏掉一个
  4. b[3]=Array();
  5. b[3]["a"]="abc";
